\section{性能値の計算}
\begin{frame}
  \frametitle{性能値の計算}
	\begin{block}{理論性能値}
	\begin{equation*}
		%(\mbox{性能値GFlops}) = (\mbox{1Clockあたりの演算回数}) * (\mbox{Frequency}) * (\mbox{コア数}) \\
		(GFlops) = (FLOPS/Clock) * (Frequency) * (Num\_Cores)
  \end{equation*}
	\end{block}

	\begin{block}{実測性能値}
	\begin{center}
  \begin{equation*}
		(\mbox{GFlops}) = \frac{(\mbox{計算1回あたりの演算回数})*(\mbox{計算回数})}{(\mbox{全体で掛かった時間)*($10^9$)}}
  \end{equation*}
	\end{center}
	\end{block}
	Flopsとは，1秒あたりの演算回数(Flop/sec)\\
	Aluminium計算機での理論値
	\begin{equation*}
		2 * 2.93 * 1 = 5.86GFlops
  \end{equation*}
\end{frame}
